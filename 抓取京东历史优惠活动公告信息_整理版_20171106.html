# 这里定义一个函数，将网址转化成html格式
# 区分get,post模式

import requests

def url2html(url,type = 'get',encoding='utf-8'):
    # 默认设置模式为get，编码格式为 utf-8
    try:
        if type == 'get':
            r = requests.get(url,timeout=120)
        elif type == 'post':
            r = requests.post(url,timeout=120)
    except requests.exceptions.ConnectTimeout:
        print('Wrong URL! Please Check')
    else:
        if r.status_code == 200:
            r.encoding = encoding
            content = r.text
        return content
# 要抓取的网址的格式如下： # https://www.jd.com/queryNews.html?type=2&title=&page=3' # 其中参数解释如下 # type = 2 就是【活动促销】栏目下的，信息，type=1是【京东公告】下的信息 # page = 3 这个就是页面序号

df_res = pd.DataFrame()

# 页面序号作为循环变量
# 一共650个页面，这里使用range函数
# range函数有“左闭右开”的特征，及range(1,N)得到的序列长度为N-1
for page_id in range(1,651):
    # 每次抓取的网页地址
    url = 'https://www.jd.com/queryNews.html?type=2&title=&page=' + str(page_id)
    
    # 将网址转化为html
    html = url2html(url)
    
    # 将txt格式的html转化为json格式，即python中的dict格式
    html_json = json.loads(html)
    
    # 将json格式(或者dict格式)的数据转化为dataframe
    df_tmp = pd.DataFrame(html_json['record'])
    
    # 和保存数据的dataframe合并
    df_res = pd.concat([df_res,df_tmp])

# 分组bar
# https://chrisalbon.com/python/matplotlib_grouped_bar_plot.html
import matplotlib
# 提取颜色库
# https://stackoverflow.com/questions/22408237/named-colors-in-matplotlib
# http://www.cnblogs.com/darkknightzh/p/6117528.html
colorset = [hex for name, hex in matplotlib.colors.cnames.items() if 'FF'  in hex and '00'  in hex]

import random
# 用于颜色随机抽样

def groupbar(df,idx,cols,title = 'GroupBarRes',xlabel = 'x label',ylabel='y label'):
    # Setting the positions and width for the bars
    col_cnt = len(cols)
    colors = random.sample(colorset,col_cnt)
    
    pos = list(range(len(df[idx]))) 
    width = 0.2

    # Plotting the bars
    fig, ax = plt.subplots(figsize=(10,5))

    # 循环作图
    for i,col in enumerate(cols):
        plt.bar([p + width*i for p in pos],df[col],width,alpha=0.5,color = colors[i])

    # 图表标题
    ax.set_title(title)
    # Y轴标题
    ax.set_ylabel(ylabel)
    # X轴标题
    ax.set_xlabel(xlabel)
    # X轴标签位置
    ax.set_xticks([p + 1.5 * width for p in pos])
    # X轴标签
    ax.set_xticklabels(df[idx])
    # 坐标轴范围
    plt.xlim(min(pos)-width, max(pos)+width*4)
    plt.ylim([0, max([max(df[i]) for i in cols])*1.1])
    # 添加图例
    plt.legend(cols, loc='upper left')
    #plt.grid()
    plt.show()
